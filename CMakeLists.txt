cmake_minimum_required(VERSION 3.19)
project(las2land LANGUAGES CXX VERSION 0.1)
add_executable(las2land src/main.cpp)
target_compile_definitions(las2land PRIVATE VERSION="${las2land_VERSION}")
set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -Wfatal-errors -Wall -Wextra -Wpedantic")
set_target_properties(las2land PROPERTIES CXX_STANDARD 20 CXX_STANDARD_REQUIRED YES CXX_EXTENSIONS OFF)
install(TARGETS las2land RUNTIME DESTINATION bin)
